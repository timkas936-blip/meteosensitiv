    <script>
        // Configuration
        const CONFIG = {
            updateInterval: 3 * 60 * 60 * 1000, // 3 hours in milliseconds
            location: 'Kyiv', // Default location
            solarPanelCapacity: 5000, // 5 kW system
        };

        // Initialize
        document.addEventListener('DOMContentLoaded', function() {
            fetchData();
            setInterval(fetchData, CONFIG.updateInterval);
            loadMessages();
        });

        // Fetch all data
        async function fetchData() {
            try {
                await Promise.all([
                    fetchSolarActivity(),
                    fetchWeather(),
                    calculateSolarProduction()
                ]);
                updateLastUpdateTime();
            } catch (error) {
                console.error('Error fetching data:', error);
            }
        }

        // Fetch solar activity
        async function fetchSolarActivity() {
            try {
                // Fetch solar flare data
                const flareResponse = await fetch('https://services.swpc.noaa.gov/products/goes-xray-flux.json');
                const flareData = await flareResponse.json();
                const latestFlare = flareData[flareData.length - 1];
                const flareClass = latestFlare ? extractFlareClass(latestFlare[3]) : 'C1.0';
                
                // Fetch Kp index
                const kpResponse = await fetch('https://services.swpc.noaa.gov/products/noaa-planetary-k-index.json');
                const kpData = await kpResponse.json();
                const latestKp = kpData[kpData.length - 1];
                const kpValue = latestKp ? latestKp[3] : '3.0';
                
                // Update UI
                document.getElementById('flareClass').textContent = flareClass;
                document.getElementById('kpIndex').textContent = 'Kp ' + kpValue;
                
                // Update statuses
                updateFlareStatus(flareClass);
                updateGeomagStatus(kpValue);
                
                // Check for alerts
                if (parseFloat(kpValue) >= 6 || flareClass.startsWith('M') || flareClass.startsWith('X')) {
                    document.getElementById('alertBanner').style.display = 'block';
                }
                
            } catch (error) {
                console.error('Error fetching solar activity:', error);
            }
        }

        // Extract flare class
        function extractFlareClass(value) {
            if (!value) return 'C1.0';
            const match = value.match(/[ABCXM]\d+\.\d+/);
            return match ? match[0] : 'C1.0';
        }

        // Update flare status
        function updateFlareStatus(flareClass) {
            const statusEl = document.getElementById('flareStatus');
            if (flareClass.startsWith('X')) {
                statusEl.className = 'status-badge status-extreme';
                statusEl.textContent = 'ЕКСТРЕМАЛЬНО';
            } else if (flareClass.startsWith('M')) {
                statusEl.className = 'status-badge status-high';
                statusEl.textContent = 'ВИСОКА';
            } else if (flareClass.startsWith('C')) {
                statusEl.className = 'status-badge status-moderate';
                statusEl.textContent = 'ПОМІРНА';
            } else {
                statusEl.className = 'status-badge status-low';
                statusEl.textContent = 'НИЗЬКА';
            }
        }

        // Update geomagnetic status
        function updateGeomagStatus(kpValue) {
            const statusEl = document.getElementById('geomagStatus');
            const kp = parseFloat(kpValue);
            if (kp >= 9) {
                statusEl.className = 'status-badge status-extreme';
                statusEl.textContent = 'G5 ЕКСТРЕМАЛЬНО';
            } else if (kp >= 8) {

        // Get weather forecast for sensitive people
        function getWeatherForecast(weather) {
            let advice = [];
            
            if (weather.temp > 28) {
                advice.push('Спека! Пийте більше води, уникайте сонця.');
            } else if (weather.temp < 5) {
                advice.push('Холодно! Одягайтеся тепліше.');
            }
            
            if (weather.humidity > 80) {
                advice.push('Висока вологість — можлива слабкість.');
            } else if (weather.humidity < 30) {
                advice.push('Низька вологість — зволожуйте повітря.');
            }
            
            if (weather.windSpeed > 10) {
                advice.push('Сильний вітер — обережно на вулиці.');
            }
            
            if (weather.pressure < 1000 || weather.pressure > 1025) {
                advice.push('Стрибки тиску — можливе погіршення самопочуття.');
            }
            
            return advice.length > 0 ? advice.join(' ') : 'Погода сприятлива для метеочутливих!';
        }

        // Calculate solar production
        async function calculateSolarProduction() {
            try {
                // Get current hour
                const hour = new Date().getHours();
                
                // Simple solar production model
                // Peak production at 12-14 hours
                let efficiency = 0;
                if (hour >= 6 && hour <= 20) {
                    const peakHour = 13;
                    const distanceFromPeak = Math.abs(hour - peakHour);
                    efficiency = Math.max(0, 1 - (distanceFromPeak / 8));
                }
                
                // Adjust for weather (mock)
                const weatherFactor = 0.8; // Could be fetched from weather API
                
                const currentPower = CONFIG.solarPanelCapacity * efficiency * weatherFactor;
                const dailyProduction = currentPower * 8 / 1000; // kWh
                
                document.getElementById('solarPower').textContent = Math.round(currentPower) + ' W';
                document.getElementById('solarToday').textContent = dailyProduction.toFixed(1) + ' kWh';
                
                // Update chart
                updateProductionChart();
                
                // Forecast
                const forecast = efficiency > 0.5 ? 
                    'Сьогодні хороший день для виробництва!' : 
                    'Низька активність сонця.';
                document.getElementById('solarForecast').textContent = forecast;
                
            } catch (error) {
                console.error('Error calculating solar production:', error);
            }
        }

        // Update production chart
        function updateProductionChart() {
            const chart = document.getElementById('productionChart');
            chart.innerHTML = '';
            
            // Create bars for each hour (6 AM to 8 PM)
            for (let hour = 6; hour <= 20; hour++) {
                const bar = document.createElement('div');
                bar.className = 'chart-bar';
                
                // Calculate height based on solar curve
                const peakHour = 13;
                const distanceFromPeak = Math.abs(hour - peakHour);
                const height = Math.max(10, 100 - (distanceFromPeak * 12));
                
                bar.style.height = height + '%';
                bar.title = hour + ':00';
                
                chart.appendChild(bar);
            }
        }

        // Chat functionality
        function sendMessage() {
            const input = document.getElementById('chatInput');
            const wellbeing = document.getElementById('wellbeingSelect');
            const message = input.value.trim();
            
            if (message) {
                const container = document.getElementById('chatContainer');
                const messageDiv = document.createElement('div');
                messageDiv.className = 'chat-message';
                
                const now = new Date();
                const timeStr = now.getHours().toString().padStart(2, '0') + ':' + 
                               now.getMinutes().toString().padStart(2, '0');
                
                let wellbeingClass = 'wellbeing-normal';
                let wellbeingText = 'Нормально';
                
                if (wellbeing.value === 'good') {
                    wellbeingClass = 'wellbeing-good';
                    wellbeingText = 'Чудово 😊';
                } else if (wellbeing.value === 'bad') {
                    wellbeingClass = 'wellbeing-bad';
                    wellbeingText = 'Погано 😔';
                }
                
                messageDiv.innerHTML = `
                    <div class="chat-header">
                        <span class="chat-user">Ви</span>
                        <span class="chat-time">${timeStr}</span>
                    </div>
                    <div>${message}</div>
                    <span class="chat-wellbeing ${wellbeingClass}">${wellbeingText}</span>
                `;
                
                container.appendChild(messageDiv);
                container.scrollTop = container.scrollHeight;
                
                input.value = '';
                
                // Save to localStorage
                saveMessage(message, wellbeingText);
            }
        }

        // Save message to localStorage
        function saveMessage(text, wellbeing) {
            const messages = JSON.parse(localStorage.getItem('chatMessages') || '[]');
            messages.push({
                text: text,
                wellbeing: wellbeing,
                time: new Date().toISOString(),
                user: 'Ви'
            });
            localStorage.setItem('chatMessages', JSON.stringify(messages.slice(-50))); // Keep last 50
        }

        // Load messages from localStorage
        function loadMessages() {
            const messages = JSON.parse(localStorage.getItem('chatMessages') || '[]');
            const container = document.getElementById('chatContainer');
            
            messages.forEach(msg => {
                const messageDiv = document.createElement('div');
                messageDiv.className = 'chat-message';
                
                const date = new Date(msg.time);
                const timeStr = date.getHours().toString().padStart(2, '0') + ':' + 
                               date.getMinutes().toString().padStart(2, '0');
                
                let wellbeingClass = 'wellbeing-normal';
                if (msg.wellbeing.includes('Чудово')) wellbeingClass = 'wellbeing-good';
                if (msg.wellbeing.includes('Погано')) wellbeingClass = 'wellbeing-bad';
                
                messageDiv.innerHTML = `
                    <div class="chat-header">
                        <span class="chat-user">${msg.user}</span>
                        <span class="chat-time">${timeStr}</span>
                    </div>
                    <div>${msg.text}</div>
                    <span class="chat-wellbeing ${wellbeingClass}">${msg.wellbeing}</span>
                `;
                
                container.appendChild(messageDiv);
            });
            
            container.scrollTop = container.scrollHeight;
        }
